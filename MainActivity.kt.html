48   @Composable
49   fun MessageCard(msg: Message) {
50       // Add padding around our message
51       Row(modifier = Modifier.padding(all = 8.dp)) {
52           Image(
53               painter = painterResource(R.drawable.profile_pic),
54               contentDescription = "Contact profile picture",
55               modifier = Modifier
56                   // Set image size to 40 dp
57                   .size(40.dp)
58                   // Clip image to be shaped as a circle
59                   .clip(CircleShape)
60           )
61   
62           // Add a horizontal space between the image and the column
63           Spacer(modifier = Modifier.width(8.dp))
64   
65           Column {
66               Text(text = msg.author)
67               // Add a vertical space between the author and message texts
68               Spacer(modifier = Modifier.height(4.dp))
69               Text(text = msg.body)
70           }
71       }
72   }
